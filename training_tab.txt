            // Ticker selection
            ui.horizontal(|ui| {
                ui.label("Select Asset:");
                egui::ComboBox::from_label("")

                    .show_ui(ui, |ui| {
                        ui.selectable_value(&mut self.training_ticker, "BTC".to_string(), "Bitcoin (BTC)");

                    });
            });
            
            ui.add_space(10.0);
            
            // Training controls
            ui.horizontal(|ui| {
                if self.training_in_progress {
                    ui.spinner();
                    ui.label("Training in progress...");
                    ui.label(format!("{:.0}%", self.training_progress * 100.0));
                } else {
                    if ui.button("ðŸš€ Start Training").clicked() {

                    }
                    if ui.button("ðŸ“¦ Mint NFT from Last Model").clicked() {

            // Training info

            ui.label("â€¢ Fetch historical price data");
            ui.label("â€¢ Train neural network");
            ui.label("â€¢ Run paper trading backtest");
            ui.label("â€¢ Generate performance metrics");
            ui.label("â€¢ Auto-mint NFT on completion");
        });
        
        ui.add_space(10.0);
        ui.separator();
        
        // Training logs
        ui.heading("Training Logs");
        egui::ScrollArea::vertical()
            .max_height(300.0)
            .show(ui, |ui| {
                if self.training_logs.is_empty() {
                    ui.label("No training logs yet. Start training to see progress.");
                } else {
                    ui.label(&self.training_logs);
                }
            });
    }

        self.training_in_progress = true;
        self.training_progress = 0.0;
        self.training_logs = format!("Starting training for {}...\n", self.training_ticker);
